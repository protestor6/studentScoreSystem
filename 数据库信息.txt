数据库名：SSSdatabase
关系模式
(1） 院系department（院系号deptno，院系名称deptname）
(2） 专业major（专业号mno，专业名称mname，院系号deptno）
(3） 班级 class（班级号 classno，班级名称 classname，年级 grade，专业号 mno，辅导员id tno）
(4） 学生student（学号sno，姓名sname，性别sex，入学时间enroll_time，身份证号sid，班级号classno）
(5） 教师teacher（教师编号tno，姓名tname，性别sex，身份证号tid，职称title，院系号deptno，管理员编号admno）
(6） 课程course（课程编号cno，课程名称cname，课程类型ctype，学分credit，学时period，考核形式testmethod）
(7） 授课teach_course（授课id tcid，课程号cno，教师号tno，班级号classno）
(8） 成绩score（成绩id scoreid，平时成绩usual_score，期末成绩exam_score，总评成绩total_score，成绩等级score_grade，绩点gpa，状态status，学号sno，授课id tcid）
(9） 管理员admin（管理员编号admno，教师编号tno）
(10）账号users（用户id uid，用户类型utype，密码password，用户编号uno，管理员编号admno，连续错误次数error_count，锁定开始时间lock_time，锁定时长lock_minutes，永久锁定is_permanent_lock）
(11）课程-专业major_course（课程-专业id mcid，课程号cno，专业号mno，开课学期semester）

相关mysql语句：
# 建库
CREATE DATABASE IF NOT EXISTS SSSdatabase DEFAULT CHARSET utf8mb4;
USE SSSdatabase;

#1.院系表
CREATE TABLE department (
    deptno INT AUTO_INCREMENT PRIMARY KEY COMMENT '院系唯一标识(自增主键)',
    deptname VARCHAR(50) NOT NULL COMMENT '院系全称'
) COMMENT '院系信息表';

#2.专业表
CREATE TABLE major (
    mno INT AUTO_INCREMENT PRIMARY KEY COMMENT '专业唯一标识(自增主键)',
    mname VARCHAR(50) NOT NULL COMMENT '专业名称',
    deptno INT NOT NULL COMMENT '开设该专业的院系号',
    FOREIGN KEY (deptno) REFERENCES department(deptno)
) COMMENT '专业信息表';

#3.教师表
CREATE TABLE teacher (
    tno VARCHAR(20) PRIMARY KEY COMMENT '教师唯一标识',
    tname VARCHAR(20) NOT NULL COMMENT '教师姓名',
    sex ENUM('男','女') NOT NULL COMMENT '性别',
    tid VARCHAR(18) UNIQUE COMMENT '身份证号，唯一',
    title ENUM('教授','副教授','讲师','助教') NOT NULL COMMENT '职称',
    deptno INT NOT NULL COMMENT '教师所在院系',
    admno INT COMMENT '管理该教师的管理员编号，管理员自己为null',
    FOREIGN KEY (deptno) REFERENCES department(deptno)
) COMMENT '教师信息表';

#4.班级表
CREATE TABLE class (
    classno INT AUTO_INCREMENT PRIMARY KEY COMMENT '班级唯一标识(自增主键)',
    classname VARCHAR(30) NOT NULL COMMENT '班级名称，如软件24-1班',
    grade VARCHAR(4) COMMENT '年级，如2024',
    mno INT NOT NULL COMMENT '班级所属专业',
    tno VARCHAR(20) COMMENT '负责该班级的辅导员编号',
    FOREIGN KEY (mno) REFERENCES major(mno),
    FOREIGN KEY (tno) REFERENCES teacher(tno)
) COMMENT '班级信息表';

#5.学生表
CREATE TABLE student (
    sno VARCHAR(12) PRIMARY KEY COMMENT '学生唯一标识',
    sname VARCHAR(20) NOT NULL COMMENT '学生姓名',
    sex ENUM('男','女') NOT NULL COMMENT '性别',
    enroll_time DATE COMMENT '入学日期',
    sid VARCHAR(18) UNIQUE COMMENT '身份证号，唯一',
    classno INT NOT NULL COMMENT '学生所在班级',
    FOREIGN KEY (classno) REFERENCES class(classno)
) COMMENT '学生信息表';

#6.管理员表
CREATE TABLE admin (
    admno INT AUTO_INCREMENT PRIMARY KEY COMMENT '管理员唯一标识(自增主键)',
    tno VARCHAR(20) NOT NULL COMMENT '该管理员教师的编号',
    FOREIGN KEY (tno) REFERENCES teacher(tno)
) COMMENT '管理员信息表';

#7.课程表
CREATE TABLE course (
    cno VARCHAR(20) PRIMARY KEY COMMENT '课程唯一标识',
    cname VARCHAR(50) NOT NULL COMMENT '课程名称',
    credit FLOAT COMMENT '课程学分',
    period INT COMMENT '总学时',
    ctype ENUM('必修','选修','公选') NOT NULL COMMENT '课程类型',
    testmethod ENUM('考试','考察') NOT NULL COMMENT '考核形式'
) COMMENT '课程信息表';

#8.授课表
CREATE TABLE teach_course (
    tcid INT AUTO_INCREMENT PRIMARY KEY COMMENT '授课信息唯一标识(自增主键)',
    tno VARCHAR(20) NOT NULL COMMENT '教师号',
    cno VARCHAR(20) NOT NULL COMMENT '课程号',
    classno INT NOT NULL COMMENT '班级号',
    FOREIGN KEY (tno) REFERENCES teacher(tno),
    FOREIGN KEY (cno) REFERENCES course(cno),
    FOREIGN KEY (classno) REFERENCES class(classno),
    UNIQUE (tno, cno, classno) COMMENT '防止同一老师同一课程同一班级重复授课'
) COMMENT '授课信息表';
#9.成绩表
CREATE TABLE score (
    scoreid INT AUTO_INCREMENT PRIMARY KEY COMMENT '成绩唯一标识(自增主键)',
    sno VARCHAR(12) NOT NULL COMMENT '此成绩对应的学生',
    tcid INT NOT NULL COMMENT '此成绩对应的授课信息',
    usual_score FLOAT COMMENT '平时成绩，0-100，仅考试课有效',
    exam_score FLOAT COMMENT '期末成绩，0-100，仅考试课有效',
    total_score DECIMAL(5,1) GENERATED ALWAYS AS (usual_score*0.4 + exam_score*0.6) STORED COMMENT '总评成绩',
    score_grade ENUM('优','良','中','及格','不及格') COMMENT '成绩等级：仅考查课有效',
    gpa DECIMAL(4,2) GENERATED ALWAYS AS (
        CASE
            WHEN total_score >= 90 THEN 4.00
            WHEN total_score >= 85 THEN 3.50
            WHEN total_score >= 80 THEN 3.00
            WHEN total_score >= 75 THEN 2.50
            WHEN total_score >= 70 THEN 2.00
            WHEN total_score >= 65 THEN 1.50
            WHEN total_score >= 60 THEN 1.00
            ELSE 0.00
        END
    ) STORED COMMENT '绩点',
    status ENUM('正常','审核中','驳回') NOT NULL COMMENT '状态',
    FOREIGN KEY (sno) REFERENCES student(sno),
    FOREIGN KEY (tcid) REFERENCES teach_course(tcid)
) COMMENT '成绩信息表';
#10.账号表
CREATE TABLE users (
    uid INT AUTO_INCREMENT PRIMARY KEY COMMENT '用户唯一标识(自增主键)',
    utype ENUM('学生','教师','管理员') NOT NULL COMMENT '用户类型',
    uno VARCHAR(20) NOT NULL COMMENT '用户编号：学生/教师号',
    password VARCHAR(72) NOT NULL COMMENT '密码(bcrypt加密)',
    admno INT COMMENT '管理该账号的管理员编号',
    error_count INT DEFAULT 0,         -- 连续错误次数
    lock_time DATETIME NULL DEFAULT NULL,           -- 锁定开始时间
    lock_minutes INT DEFAULT 0,        -- 锁定时长（分钟）
    is_permanent_lock TINYINT DEFAULT 0, -- 1=永久锁定 0=未永久锁定
    FOREIGN KEY (admno) REFERENCES admin(admno)
) COMMENT '账号信息表';

#11.课程-专业表
CREATE TABLE major_course (
    mcid INT AUTO_INCREMENT PRIMARY KEY COMMENT '专业-课程唯一标识(自增主键)',
    cno VARCHAR(20) NOT NULL COMMENT '课程号',
    mno INT NOT NULL COMMENT '专业号',
    semester VARCHAR(20) COMMENT '开课学期，如2024-2025-1',
    FOREIGN KEY (cno) REFERENCES course(cno),
    FOREIGN KEY (mno) REFERENCES major(mno)
) COMMENT '课程-专业关联表';

#我们需要创建一个用户连接数据库与后端：

-- 只授予：查询、插入、更新、删除 四张基础权限
GRANT SELECT,INSERT,UPDATE,DELETE ON SSSdatabase.* TO 'user'@'%';

-- 刷新权限生效
FLUSH PRIVILEGES;

-- 加索引
-- 给总评成绩加索引，大幅提升排序速度
CREATE INDEX idx_score_total_score ON score(total_score);
-- 给学生学号建立索引，视图查询极速优化
CREATE INDEX idx_student_sno ON student(sno);
-- 额外：成绩表学号也加上，双重提速
CREATE INDEX idx_score_sno ON score(sno);

-- 创建视图方便学生成绩查询
CREATE VIEW v_student_score_detail AS
SELECT 
    s.sno             AS 学号,
    s.sname           AS 学生姓名,
    mc.semester       AS 开课学期,
    c.cno             AS 课程编号,
    c.cname           AS 课程名称,
    CASE 
        WHEN c.testmethod = '考试' THEN sc.total_score
        WHEN c.testmethod = '考察' THEN sc.score_grade
        ELSE '无成绩'
    END               AS 成绩,
    c.credit          AS 学分,
    c.period          AS 总学时,
    sc.gpa            AS 绩点,
    c.testmethod      AS 考核方式,
    c.ctype           AS 课程类型
FROM score sc
JOIN teach_course tc   ON sc.tcid = tc.tcid
JOIN course c          ON tc.cno = c.cno
JOIN student s         ON sc.sno = s.sno
JOIN class cl          ON s.classno = cl.classno
JOIN major_course mc   ON c.cno = mc.cno AND cl.mno = mc.mno; 

-- 创建视图方便学业统计查询

-- 新建学生统计总视图（包含：学号、专业号、总绩点、平均绩点、已修学分、专业总学分、未修学分）
-- IFNULL(表达式，0)是指表达式是NULL时赋值为0，否则为其原本的值
CREATE VIEW v_student_stat_detail AS
SELECT
    s.sno                 AS 学号,             -- 学生号
    cl.mno                AS 专业号,             -- 专业号
    IFNULL(SUM(sc.gpa), 0)          AS 个人总绩点,        -- 个人总绩点
    IFNULL(SUM(sc.gpa * c.credit) / SUM(c.credit), 0.0)          AS 平均学分绩点,          -- 平均学分绩点
    IFNULL(SUM(c.credit), 0)        AS 已修学分,-- 已修学分

    -- 本专业应修总学分（动态从 major_course + course 计算）
    (
        SELECT IFNULL(SUM(c2.credit), 0)
        FROM major_course mc2
        JOIN course c2 ON mc2.cno = c2.cno
        WHERE mc2.mno = cl.mno
    )                       AS 总学分,     -- 总学分（动态）

    -- 未修学分 = 总学分 - 已修学分
    (
        SELECT IFNULL(SUM(c2.credit), 0)
        FROM major_course mc2
        JOIN course c2 ON mc2.cno = c2.cno
        WHERE mc2.mno = cl.mno
    ) - IFNULL(SUM(c.credit), 0)  AS 未修学分 -- 未修学分

FROM student s
JOIN class cl ON s.classno = cl.classno
LEFT JOIN score sc ON s.sno = sc.sno
LEFT JOIN teach_course tc ON sc.tcid = tc.tcid
LEFT JOIN course c ON tc.cno = c.cno

GROUP BY s.sno, cl.mno;
